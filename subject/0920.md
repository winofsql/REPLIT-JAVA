### [Java の外部クラス](https://replit.com/@sworc/java-0920-1)
```java
import java.util.Scanner;
import java.util.*;

class Main {

  
  // ******************************
  // ✅ 外部クラス
  // ******************************
  public static Cfor v_for;
  public static Cforex v_forex;
  
  // ******************************
  // ✅ 最初のインスタンス作成
  // ******************************
  public static List<String> rpg = new ArrayList<String>();
  
  public static void main(String[] args) {

    // ******************************
    // ✅ 追加( 初期値設定 )
    // ******************************
    rpg.add("3:勇者");
    rpg.add("2:賢者");
    rpg.add("1:魔導士");    

    
    System.out.println("【q を入力すると終了します】");

    Scanner scan = new Scanner(System.in);

    String str = "";
    while( !str.equals("q") ) {
      System.out.println();
      System.out.print("system >> ");

      str = scan.nextLine();
      System.out.println(str);

      // ******************************
      // 🔶 一般的な for
      // ******************************
      if ( str.equals("for") ) {
        v_for = new Cfor();
        v_for.action( rpg );
      }

      // ******************************
      // 🔶 拡張 for ( 他言語の foreach )
      // ******************************
      if ( str.equals("forex") ) {
        v_forex = new Cforex();
        v_forex.action( rpg );
      }      
      
    }

    scan.close();
    
    System.out.println();
    System.out.println("【🔴 処理を終了しました】\n");
    
  }
}
```
